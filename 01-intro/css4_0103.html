Printf
printf ("[%s]" , 'monkey');//[monkey]
printf ("[%10s]" , 'monkey'); // [ monkey]
printf ("[%-10s]" , 'monkey'); // [monkey ]
printf ("[%010s]" , 'monkey');//[0000monkey]
printf ("[%'#10s]" , 'monkey');//[####monkey]
printf ('%.2f' , 27.452); // 27.45
printf ('%.2f%% Complete' , 2.1); // 2.10% Complete

 There are four ways to send output to the browser

print_r() // useful for arrays, objects, and other things
