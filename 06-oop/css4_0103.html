<?php


class Pirovneba {
  //properties
  public $fname;
  public $lname;

//methods

function __construct($name, $surname) {
  $this->fname = $name;
  $this->lname = $surname;
}


public function intro() {
  echo "My name is {$this->fname} and surname is {$this->lname}.";
}
}

class Patriot extends Pirovneba {
  public function message() {
    echo "Am I a patriot or not? ";
  }

}


$myself = new Patriot("tsotne", "meladze");
echo $myself->message(); //Am I a patriot or not?



?>
